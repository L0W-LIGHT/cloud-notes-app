name: Deploy App

on:
  push:
    branches:
      - main

jobs:
  deploy:

    runs-on: ubuntu-latest

    steps:

      - name: Deploy to EC2

        uses: appleboy/ssh-action@master

        with:
          host: ${{ secrets.EC2_HOST }}
          username: ${{ secrets.EC2_USERNAME }}
          key: ${{ secrets.EC2_SSH_KEY }}

          script: |

            cd cloud-notes-app

            git pull

            docker stop mycontainer || true
            docker rm mycontainer || true

            docker build -t myapp .

            docker run -d \
              --name mycontainer \
              -p 80:80 \
              myapp
